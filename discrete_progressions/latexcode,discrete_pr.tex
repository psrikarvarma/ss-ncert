\section*{Question:}

Then from table of parameters,
\begin{align}

\begin{align}

 \item \begin{align}

\end{align}
\item \begin{align}

\end{align}
\item \begin{align}
    x_i\brak{n} &=
    \begin{cases}
        \brak{\frac{n+2}{2}}u\brak{n} & \text{if } i=1 \\
        \brak{\frac{10-n}{2}}u\brak{n} & \text{if } i=2 
    \end{cases}
\end{align}
\end{enumerate}
